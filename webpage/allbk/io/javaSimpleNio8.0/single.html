package test3;
import java.util.concurrent.CompletableFuture;
import java.util.concurrent.ExecutionException;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

public class ComFutureTest {
public static void main(String[] args) {
//创建线程池
ExecutorService executor = Executors.newFixedThreadPool(10);
// trigger cupd 创建一个实现Supplier接口类
ComFutureTest1 comFutureTest1 = new ComFutureTest1();
//CompletableFuture可以异步取值 （异步执行）
CompletableFuture cupdResult = CompletableFuture.supplyAsync(comFutureTest1,executor);
//关闭线程池 线程池不会立马关闭只会不让在创建线程任务 在执行的线程执行完后会自动关闭
executor.shutdown();
// return cupd 创建一个实现Consumer接口类
ComFurureTest2 comFurureTest2 = new ComFurureTest2();
cupdResult.thenAccept(comFurureTest2);
try {
//使用get会阻塞主线程过去值 System.out.println(cupdResult.get());
} catch (InterruptedException e) {
// TODO Auto-generated catch block
e.printStackTrace();
} catch (ExecutionException e) {
// TODO Auto-generated catch block e.printStackTrace();
}
// return RM
System.out.println("结束mian方法 return RM msg to customer");
}
}
