package jdbc;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
public class ComJDBC {
//数据库驱动
private final String command = "com.mysql.jdbc.Driver";
//连接数据库
private final String url="jdbc:mysql://localhost:3306/databasesTest";
//连接数据库账号
private final String userName = "root";
//连接数据库密码
private final String password = "root";
//与数据源进行绘画
private Connection connection = null;

private PreparedStatement preparedStatement = null;
private ResultSet resultSet =null;
// 加载驱动
private void getClassName() {
try {
Class.forName(command);
} catch (ClassNotFoundException e) {
// TODO Auto-generated catch block
e.printStackTrace();
}
}
//关闭连接
public void getClose() throws SQLException{
if(resultSet!=null)
resultSet.close();
if(preparedStatement!=null)
preparedStatement.close();
if(connection!=null)
connection.close();
}
//修改
public int getUpdate(String updateName , String name) throws SQLException{
//加载驱动
getClassName();
//建立连接
connection=DriverManager.getConnection(url, userName, password);
String sql = "UPDATE my_db SET NAME=? WHERE NAME=?";
preparedStatement=connection.prepareStatement(sql);
preparedStatement.setString(1, updateName);
preparedStatement.setString(2, name);
int number = preparedStatement.executeUpdate();
return number;
}
//查询
public ResultSet getSelect() throws SQLException{
//加载驱动
getClassName();
//建立连接
connection = DriverManager.getConnection(url,userName,password);
String sql = "select * from my_db";
preparedStatement=connection.prepareStatement(sql);
resultSet = preparedStatement.executeQuery();
return resultSet;
}
//删除
public int getDetele(String name) throws SQLException{
//加载驱动
getClassName();
//建立连接
connection = DriverManager.getConnection(url,userName,password);
String sql = "delete from my_db where name=?";
preparedStatement = connection.prepareStatement(sql);
preparedStatement.setString(1, name);
int number = preparedStatement.executeUpdate();
return number;
}

}
