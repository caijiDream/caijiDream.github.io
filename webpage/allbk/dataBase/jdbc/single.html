package jdbc;

import java.sql.ResultSet;
import java.sql.SQLException;

public class JDBCTest {
public static void main(String[] args) {
ComJDBC comJDBC = new ComJDBC();
//查询
try {
//接收返回值
ResultSet resultSet= comJDBC.getSelect();
//resultSet类似于指针直接指向下一行即可获取第一行结果值（也可以理解为resultset的第一行为null要从下一行开始）
while (resultSet.next()) {
//name为返回数据字段 是什么类型get什么
System.out.println(resultSet.getString("name"));
}
//使用完后要关闭连接数据库节约资源
comJDBC.getClose();
} catch (SQLException e) {
// TODO Auto-generated catch block
e.printStackTrace();
}
//修改操作
try {
System.out.println(comJDBC.getUpdate("name1", "name修改"));
comJDBC.getClose();
} catch (SQLException e) {
// TODO Auto-generated catch block
e.printStackTrace();
}

//删除操作
try {
System.out.println(comJDBC.getDetele("name3"));
comJDBC.getClose();
} catch (Exception e) {
// TODO: handle exception
}
}
}
